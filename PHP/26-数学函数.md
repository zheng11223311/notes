### 1.数学函数

```php
<?php
//abs()	绝对值 负变正 正还是正
$num=-10;
$result=abs($num);
var_dump($result);	//int 10


//ceil() 向上取整
$num=3.14;
$result=ceil($num);
var_dump($result);	//float 4

//floor() 向下取整
$num=3.14;
$result=ceil($num);
var_dump($result);	//float 3

//round() 对浮点数进行四舍五入
$num=3.14;
$result=round($num);
var_dump($result);	//float 3

$num=3.141;
$result=round($num,2);	//保留2 位小数
var_dump($result);	//float 3.14


//max() 找出最大值
$max=max(1,2,3,4,321,'333hello');
var_dump($max);	//333hello,去掉字母后比较

$max=max(array(1,2,3,4,321,'333hello'));
var_dump($max);	//333hello,去掉字母后比较

//min() 找出最小值
$min=min(0,'hello',1,2,3,4,321);
var_dump($min);	//0

$min=min('hello',0,1,2,3,4,321);
var_dump($min);	//hello  纯字符串会被转化为0,谁在前就先选谁

$min=min(array(1,2,3,4,321,'333hello'));
var_dump($min);	//333hello,去掉字母后比较

//mt_round()  产生一个更好的随机数,推荐使用,因为他比round() 快4倍
//round()	产生一个随机数
var_dump(mt_round());	//返回0~mt_getrandmax() (整型的最大值)之间的值
var_dump(mt_round(1,100));	//返回[1,100]之间的值

//sqrt 平方根
var_dump(sqrt(9));	//3

//pi() 得到圆周率的值
//M_PI 得到圆周率的值
var_dump(pi());	//3.1415926
var_dump(M_PI);	//3.1415926

//pow() 指数表达式
var_dump(pow(10,2));	//100
?>
```
