```php
//htmlspecoalchars()	将特殊的字符串转为html 实体

$str=chr(65);

//A-Z	==>65-90
//a-z	==>97-122

$str='aaaa';
echo strlen($str);	//4

//chunk_split()	将字符串分割成小块
//参数1:要分割的字符串
//参数2:可选参数, 每几位分割,默认在字符串的结尾分割
//参数3:可选参数,分割符号,默认使用换行分割
//返回值:返回分割好的字符串
$str='aaaa';
echo chunk_split($str);	//aaaa+换行,以整个字符串为边界切割
echo chunk_split($str,3);	//aaa a,以3个字符为界进行分割
echo chunk_split($str,3,"-");	//aaa-a, 以3个字符为界进行分割,以- 为界限

//explode() 字符串切割
//参数1:要使用的切割符
//参数2:要切割的字符串
//返回值:返回切割后的数组
$Arr=explode("-",chunk_split($str,3,"-"));	
var_dump($Arr);	//[aaa,a]

//implode()	使用指定字符串连接数组,组成一个字符串
//参数1:要使用的连接符
//参数2:要连接的数组
//返回值:返回拼接好的字符串
$str=implode("-",$Arr);
var_dump($str);		//aaa-a
?>
```

```php
<?php
//ltrim()	删除字符串开头的空白字符(或其他字符)
$str=' 牛叉任务';
$newStr=ltrim($str);
var_dump($newStr);	//'牛叉任务'

$str='牛叉任务';
$newStr=ltrim($str,"叉牛");
var_dump($newStr);	//'任务'	顺序跌倒都可以,只要是紧挨着的开头

//rtrim()	删除字符串结尾的空白字符(或其他字符)
$str='牛叉任务 ';
$newStr=rtrim($str);
var_dump($newStr);	//'牛叉任务'

$str='牛叉任务';
$newStr=ltrim($str,"任务");
var_dump($newStr);	//'牛叉'	顺序跌倒都可以,只要是紧挨着的结尾

//trim()	删除字符串两端的空白字符(或其他字符)
//trim() ,ltrim(), rtrim() 三者的参数相同
//参数1:要删除空白字符串或其他字符的字符串
//参数2:可选参数
//指定要删除的字符,如果不设置默认删除空白字符
//返回值:返回一个新的字符串
$str=' 牛叉任务 ';
$newStr=trim($str);
var_dump($newStr);	//'牛叉任务'

$str='11牛叉任务11';
$newStr=trim($str,"11");
var_dump($newStr);	//'牛叉任务'	顺序跌倒都可以,只要是紧挨着的结尾或开头

//str_replace()	字符串替换函数
//参数1:要查找的内容 可以是字符串 可以是数组
//参数2: 要替换的内容 可以是字符串 可以是数组
//参数3:要替换内容的字符串
//参数4:可选参数 引用传参 指该函数替换的次数
$str='学习园地';
$result=str_replace("园",'',$str);

$result=str_replace($arr1,$arr2,$str);
```

```php
<?php
//substr()	截取字符串
//参数1:要截取的字符串
//参数2:从哪开始截取(字符串下标) 如果是负数表示从后往前数第几个开始,在向后截取
//参数3:可选参数 表示截取几位,如果是负数 表示保留几位
$str='aaabbbccc';
$result=substr($str,2);	//从2 开始截取 (2,结束]
var_dump($result);	//abbbccc

$result=substr($str,2,2);	//从2 开始截取,截取2位
var_dump($result);	//ab

$result=substr($str,-2);	//从倒数第2 位开始截取,向后截取 [-2,结束]
var_dump($result);	//cc

$result=substr($str,-2,1);	//从倒数第2 位开始截取,向后截取1 位	[-2,后一位]
var_dump($result);	//c
 
$result=substr($str,-3,-1);	//从倒数第3 位开始截取,保留后1位	[-2,结束-1]
var_dump($result);	//cc


//sub_count()	统计一个字符串在另一个字符串中出现的次数
//参数1:要操作的字符串
//参数2:要查找的字符串
//参数3:可选参数 表示从第几位开始查找(下标)
//参数4:可选参数 要查找多长
$str='aaabbbccc';
$new=sub_count($str,'a');
var_dump($new);	//3

$new=sub_count($str,'a',3);	//从第3 位之后开始查找 (3,结束]
var_dump($new);	//0

$new=sub_count($str,'a',3,5);	//从第3 位之后开始查找5位 (3,3+5]
var_dump($new);	//0

$new=sub_count($str,'a',-3,-1);	//
var_dump($new);	//


//strstr()	返回一个字符串在另一个字符串开始位置到结束位置的字符串
$str='aabbcc';
$new=strstr($str,'b');
var_dump($new);//bbcc

//strrch()	返回一个字符串在另一个字符串最后一次出现的位置到结束位置的字符串
$str='aabbcc';
$new=strrch($str,'b');
var_dump($new);//bcc

//strpos()	寻找字符串中某字符最先出现的位置
$str='aabbcc';
$new=strpos($str,'b');
var_dump($new);//2
//strrpos()	寻找字符串中某字符最后出现的位置
$str='aabbcc';
$new=strrpos($str,'b');
var_dump($new);//3

?>
```
