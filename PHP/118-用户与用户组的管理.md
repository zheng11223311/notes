### 1.用户与用户组的管理

```shell
vim /etc/password     #用户的信息文件
#显示
#第一行的  root:x:0:0:root:/root:/bin/bash
#root 表示用户
#x 表示密码占位,技术的不断进步,安全起见,密码以存放在别处,使用x 占位
#0 代表 uid  用户的id
#0 代表组id 
#root 对前面一个root 起到解释说明的作用
#/root  代表管理员家目录的位置
#/bin/bash  代表可以使用bash 登入

#第二行到28行
#代表系统的伪用户,给系统来用
```

```shell
vim /etc/shadow   #用户密码文件(影子文件,权限为0,看得见,摸不着,不能操作)
#显示
#第一行 root:$6$17voxanBd.....:1733:99999    //root: 后面代表用户密码,再后面代表时间
#第二行到28行
#代表系统的伪用户,给系统来用
```

```shell
vim /etc/group  #组文件
#显示
#第一行 root:x:3:
#root 组名
#x 密码位
#0 组id
#3:后面为空,代表没有附加组员
#第二行到最后,为系统的组
```

### 2.相关命令

```shell
useradd  aa   #添加用户
vim /etc/password  #查看到最后一行中写入用户
cd /home 
ls 		#查看到新添加的 aa 用户

password aa  #为aa 用户设置密码
vim /etc/shadow    #查看到最后一行中写入用户加密的密码

userdel aa   #删除aa 用户,再/home 中会留下文件
userdel -r aa  #删除aa 用户,连带家目录下的信息一起删除,不会在/home 下留下文件
```

```shell
groupadd aa  #添加组
vim /etc/group   #查看到最后一行中写入组

gpasswd -a wzs aa   /#添加add 用户wzs 到组aa 里面,g为全局
gpasswd -d wzs aa   #删除del 用户wzs 

groupdel aa    #删除组aa

su -  用户名 #切换用户身份
su - root   #切换到root 用户 普通用户切换到root 需要密码,root 切换到普通用户不需要
su root		#基本同理,切换到root 用户
su 		#基本同理,切换到root 用户
su aa 	#切换到aa 用户
```
