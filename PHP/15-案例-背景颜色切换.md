### 背景颜色切换

```php
<?php
   //1.必须获取到用户选择的内容 $_GET['表单名称']
   //2.判断用户是否选择内容,如果没有选择 需要设置默认值 否者将报错
//    $bgColor=isset($_GET['bgc'])?$_GET['bgc']:'transparent';

//null 合并运算
$bgColor=$_GET['bgc']??'transparent';

//制作当选择颜色后,让选择的颜色做默认选中的效果
//连续定义针对颜色值的变化
$red=$blue=$pink=$green=$purple='';

//进行判断 ,如果用户选择是红色,将$red地变量值改为selected
if($bgColor=="red"){
    $red='selected';
}
if($bgColor=="blue"){
    $blue='selected';
}
if($bgColor=="pink"){
    $pink='selected';
}
if($bgColor=="green"){
    $green='selected';
}
if($bgColor=="purple"){
    $purple='selected';
}
?>

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta http-equiv="X-UA-Compatible" content="IE=edge">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>背景颜色切换</title>
    <style>
        body{
            background-color:<?=$bgColor?>;
        }
    </style>
</head>
<body>
    <form action="" method="get">
        <select name="bgc">
            <option value="xz">---请选择---</option>
            <!-- 一句语句里面 echo 与= 相同 -->
            <option value="red" <?php echo $red ?> >红色</option>
            <option value="blue" <?= $blue ?> >蓝色</option>
            <option value="green" <?=$green ?> >绿色</option>
            <option value="pink" <?= $pink ?> > 粉色</option>
            <option value="purple" <?=$purple ?> >紫色</option>
        </select>
        <input type="submit" value="变变变">
    </form>
</body>
</html>

```
