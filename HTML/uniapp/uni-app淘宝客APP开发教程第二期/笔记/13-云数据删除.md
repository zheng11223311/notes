# 云数据删除

```js
'use strict';

// 连接云数库,快捷代码块cdb
// 一个账号默认只有一个数据库
const db = uniCloud.database();
// 选择数据表user
const collection=db.collection('user')



exports.main = async (event, context) => {
	
	//删除数据,doc 指定_id
	let res= await collection.doc("62a420eec2cedc0001d35dcf").remove()
	console.log('删除结果 ',res);
	
	//返回数据给客户端
	return event 
};

```

# 云数据查询

```js
'use strict';

// 连接云数库,快捷代码块cdb
// 一个账号默认只有一个数据库
const db = uniCloud.database();
// 选择数据表user
const collection=db.collection('user')



exports.main = async (event, context) => {
	
	// 查询,只有执行get 才是真正的查询
	// const res=await collection.doc("62a420db5e068d00011373b0").get()
	// 条件查询
	const res=await collection.where({
		// 字段:值
		username:'团子'
	}).get()
	console.log(res);
	
	
	//返回数据给客户端
	return res 
};

```
