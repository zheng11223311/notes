### 1.多页面环境搭建

`webpack.config.js`

```js
const path=require('path')
const HtmlWebpackPlugin=require('html-webpack-plugin')

module.exports={

    entry:{		//打包成单独的 main.js 和lodash 文件
        main:{
            import:['./src/app.js','./src/app2.js'],
            dependOn:'lodash'   //依赖下方抽离打包的lodash
            filename:'chunk1/[name].js', //文件输出到chunk1 目录下
        },
        main2:{
            import:'./src/app3.js',
            dependOn:'lodash',   //依赖下方抽离打包的key, lodash 
             filename:'chunk2/[name].js', //文件输出到chunk2 目录下
        }
        //lodash:'lodash',
        lodash:{
        	import:'lodash',
        	filename:'common/[name].js',
    	}
    }
    
    plugins:{		//插件
        new HtmlWebpackPlugin({			//根据自己的模板生成index.html 文件
	    template:"./index.html",
	    title:'多页面应用',				//title,可以为任意key 名(如title1),设置 index.html 的title,需要修改index.html 的title 使用内置的变量,为:<title><%= htmlWebpackPlugin.options.title%></title>
    	inject:'body',			//定义生成的script 标签放在index.html 的哪个位置上,body/header 
    	//chunks:[],					//规定当前页面打包那些chunk(chunk 为entry 设置打包后的文件,比如mian1),[] 为所有chunk 都打包进去
    	//chunks:['mian'],	//只打包main.js 到index.html 中	
    	chunks:['mian','mian2','lodash'],
        //filename:'index.html',		//打包后输出的文件名
		filename:'chunk1/index.html',	//放在dist/chunk1/ 目录下
        publicPath:'http://www.a.com/',		//index.html 的script 引入的文件的公共路径 http://www.a.com/main.js 和 http://www.a.com/main2.js  和 http://www.a.com/lodash.js,entry 配置了filename:'chunk1/[name].js', 则http://www.a.com/chunk1/main2.js
    	}),
            
       
		new HtmlWebpackPlugin({			//根据自己的模板生成index.html 文件
	    template:"./index2.html",
	    title:'多页面应用',				
    	inject:'body',			//定义生成的script 标签放在index.html 的哪个位置上,body/header 
    	chunks:['mian','mian2'],
        filename:'index2.html',	
		publicPath:'http://www.b.com/',		//index2.html 的script 引入的文件的公共路径 http://www.b.com/main.js 和 http://www.b.com/main2.js
    	}),
	},
        output:{
            clean:true,		//清除之前打包的dist 文件夹
        }
    
    
    mode:"developement",  // 开发模式	
}
```

执行打包

```
npx webpack
```

执行热更新

```
npx webpack server
```

浏览器:http://localhost:8080 查看
