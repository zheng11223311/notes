### 1.entry

`webpack.config.js`

```js
const path=require('path')
const HtmlWebpackPlugin=require('html-webpack-plugin')

module.exports={
    //entry:'./src/app.js',		//入口文件
    //entry:['./src/app.js','./src/app2.js'],		//多入口配置,按顺序先引入app,在引入app2
    //entry:['./src/app.js','./src/app2.js','lodash'],	//也可以加载npm 包,比如lodsah
    //entry:{		//打包成单独的 main.js 和lodash 文件
    //    main:['./src/app.js','./src/app2.js'],
    //    lodash:'lodash',
    //}
    
    //如果在app.js 直接使用import 会导致lodash 也打包到app.js 中,使文件增大,使用dependOn 就不会打包到app.js 中
    entry:{		//打包成单独的 main.js 和lodash 文件
        main:{
            import:['./src/app.js','./src/app2.js'],
            dependOn:'lodash'   //依赖下方抽离打包的lodash
        },
        main2:{
            import:'./src/app3.js',
            dependOn:'lodash'   //依赖下方抽离打包的key, lodash 
        }
        lodash:'lodash',
    }
    
    plugins:{		//插件
        new HtmlWebpackPlugin({			//根据自己的模板生成index.html 文件
	    template:"./index.html",
	    title:'多页面应用',				//title,可以为任意key 名(如title1),设置 index.html 的title,需要修改index.html 的title 使用内置的变量,为:<title><%= htmlWebpackPlugin.options.title%></title>
    inject:'body',			//定义生成的script 标签放在index.html 的哪个位置上,body/header 
    //chunks:[],					//规定当前页面打包那些chunk(chunk 为entry 设置打包后的文件,比如mian1),[] 为所有chunk 都打包进去
    //chunks:['mian'],	//只打包main.js 到index.html 中	
    chunks:['mian','mian2','lodash'],
    	}),
	},
    
    
    mode:"developement",  // 开发模式	
}
```

执行打包

```
npx webpack
```

执行热更新

```
npx webpack server
```

浏览器:http://localhost:8080 查看
